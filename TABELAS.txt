CREATE TABLE PRACAS(
ID INT PRIMARY KEY IDENTITY,
VAGA CHAR(1)NOT NULL,
ENDERECO INT NOT NULL
);

CREATE TABLE VEICULO (
ID INT PRIMARY KEY IDENTITY,
PLACA VARCHAR(6) NOT NULL,
COR VARCHAR(10),
TIPO INT NOT NULL
);


CREATE TABLE MOVIMENTACAO(
ID INT PRIMARY KEY IDENTITY,
DATAINICIO DATE,
DATAFINAL DATE,
VEICULO INT,
OCUPADO CHAR(1),
P INT,
CONSTRAINT [chack_OCUPADO] CHECK([OCUPADO] IN ('S','N')),
FOREIGN KEY (VEICULO) REFERENCES VEICULO(ID),
FOREIGN KEY (P) REFERENCES PRACAS(ID)
);
